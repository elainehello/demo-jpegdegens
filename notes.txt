# Installs development dependencies for Hardhat and Ethereum smart contract testing:
# - @nomiclabs/hardhat-ethers: Hardhat plugin for ethers.js integration
# - ethers: Ethereum JavaScript library for interacting with the blockchain
# - @nomiclabs/hardhat-waffle: Hardhat plugin for Waffle testing framework
# - ethereum-waffle: Library for writing and testing smart contracts
# - chai: Assertion library for testing

    RUN:
>$  yarn add -D @nomiclabs/hardhat-ethers ethers @nomiclabs/hardhat-waffle ethereum-waffle chai

# Installs TypeScript support for Hardhat projects:
# - ts-node: TypeScript execution environment for Node.js
# - typescript: TypeScript language support

    RUN:
>$  yarn add -D ts-node typescript

# Installs type definitions for Node.js, Mocha, and Chai (for TypeScript support in tests)

    RUN:
>$  yarn add -D chai @types/node @types/mocha @types/chai

# Pretty-prints the JSON output of the compiled HelloWorld contract using jq for easier reading

    RUN:
>$  cat artifacts/contracts/HelloWorld.sol/HelloWorld.json | jq

# Starts a local Hardhat Ethereum node for development and testing.
# This command will deploy an Ethereum network on your local computer.
# WARNING: The accounts and private keys shown are for local use only and are publicly known.
# Never use these accounts or keys on mainnet or any public network.

    RUN:
>$  npx hardhat node

# Run the deploy script from the /scripts directory using the localhost network

    RUN:
>$  npx hardhat run scripts/deploy-hello.ts --network localhost
